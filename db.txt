
CREATE TABLE reg_event(
	id_reg INT(9) AUTO_INCREMENT NOT NULL,
	id_usuario_reg INT(9),
	arbol VARCHAR(50),
	area VARCHAR(255),
	especie VARCHAR(100),
	trabajo_realizar VARCHAR(255),
	altura FLOAT(8,4),
	diametro FLOAT(8,4),
	ubicacion VARCHAR(255),
	colonia VARCHAR(255),
	PRIMARY KEY (id_reg)	
);
